tree("Root")
	sequence
		tree("ResetVariables")
		tree("Idle")
		tree("ChaseTarget")

tree("ResetVariables")
	sequence
		ResetVariables

tree("Idle")
	sequence
		not IsChasingTarget
		CheckPlayer

tree("ChaseTarget")
	sequence
		race
			not repeat RequestPathAndGoToTarget
			CheckIfReachedTarget
		tree("CaughtPlayer")

tree ("CaughtPlayer")
	sequence
		HoldPlayer
		Wait 3.0
		ReleasePlayer
		GoToStartPosition
		CheckIfReachedTarget
