tree("Root")
	sequence 
		tree("Idle")
		tree("ChaseTarget")
		tree("CaughtPlayer")

tree("Idle")
	sequence
		not IsChasingTarget
		CheckPlayer

tree("ChaseTarget")
	sequence
		race
			not repeat RequestPathAndGoToTarget
			CheckIfReachedTarget

tree ("CaughtPlayer")
	sequence
		HoldPlayer
		Wait 3.0
		ReleasePlayer
		race
			not repeat GoToStartPosition
			CheckIfReachedTarget
