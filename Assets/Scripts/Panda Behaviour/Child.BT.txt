tree("Root")
	sequence
		tree("Idle")
		tree("ChaseTarget")
		tree("CaughtPlayer")
		tree("ResetVariables")
		Wait(3.0)
		Fail

tree("ResetVariables")
	sequence
		ResetVariables

tree("Idle")
	CheckPlayer

tree("ChaseTarget")
	while IsPlayerNear
		while not IsCaptured
			sequence
				race
					not repeat RequestPathAndGoToTarget
					while IsChasingTarget
						race
							not repeat CheckPlayer
							CheckIfReachedTarget

tree ("CaughtPlayer")
	sequence
		HoldPlayer
		Wait 3.0
		ReleasePlayer
		ChangeTarget "startPos"
		race
			not repeat RequestPathAndGoToTarget
			while IsChasingTarget
				CheckIfReachedTarget
